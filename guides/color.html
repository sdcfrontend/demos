CSS Vars

/* COLORS */

/* neutral shades (for internal use) */
--shade-0: #FFFFFF;
--shade-1: #F4F5F7;
--shade-2: #E2E2E2;
--shade-3: #707070;
--shade-4: #4A4A4A;
--shade-5: #222222;
--shade-6: #000000;

/* gradients */
--gradient-linear: linear-gradient(180deg, #F5F5F5, #E3E4E4);
--gradient-radial: radial-gradient(circle at center,#fff 0,#e8e7ed 100%);
--gradient-pipe: linear-gradient(90deg,#d7d5dd 0,#eee 30%,#eee 70%,#d7d5dd);

/* ui states */
--success: #299934; /* success messages */
--alert: #EA021A; /* alerts and warnings */
--live: #EA021A; /* live coverage/scores */
--breaking: #FFE000; /* breaking news components */
--link-color: #3157A1; /* global text link color */

--live-bg: linear-gradient(180deg, #E90E10, #BC0000) /* gradient version of live */
--breaking-bg: linear-gradient(180deg, #F8EB1F, #F2CA06); /* gradient version of breaking */

/* common alias values */
--body-bg: var(--shade-1); /* mobile site bg (desktop uses a gradient) */
--component-bg: var(--shade-0); /* components bg is generally white */
--component-shade: var(--shade-1); /* component header bg */
--border-color: var(--shade-2); /* border color for components */
--text-color: var(--shade-4); /* main body text */
--text-color-invert: var(--shade-0); /* text on dark backgrounds */
--text-color-secondary: var(--shade-3); /* lighter body text */
--text-color-tertiary: #9F9F9F; /* lightest body text */
--text-color-dark: var(--shade-5); /* darker body text */
--icon-color: var(--shade-5); /* color for svg/ui icons */

/* box shadows */
--drop-shadow: 0 3px 3px rgba(0, 0, 0, 0.15); /* horizontal drop shadow beneath a ui element */
--glow-shadow: 0 0 12px rgba(0,0,0,0.15); /* glow shadow around a ui element */

/* overlays */
--black-op40: rgba(0, 0, 0, 0.4); /* a black overlay at 40% opacity */
--black-op60: rgba(0, 0, 0, 0.6); /* a black overlay at 60% opacity */
--black-op80: rgba(0, 0, 0, 0.8); /* a black overlay at 80% opacity */

CSS Vars

/* COLORS */

/* neutral shades (for internal use) */
--shade0: #FFFFFF;
--shade1: #F4F5F7;
--shade2: #E2E2E2;
--shade3: #707070;
--shade4: #4A4A4A;
--shade5: #222222;

/* gradients */
--gradient-radial: radial-gradient(circle at center,#fff 0,#e8e7ed 100%);
--gradient-pipe: linear-gradient(90deg,#d7d5dd 0,#eee 30%,#eee 70%,#d7d5dd);

/* ui states */
--success: #299934; /* success messages */
--alert: #EA021A; /* alerts and warnings */
--breaking: #FFE000; /* breaking news components */
--link-color: #3157A1; /* global text link color */

/* common alias values */
--body-bg: var(--shade1); /* mobile site bg (desktop uses a gradient) */
--component-bg: var(--shade0); /* components bg is generally white */
--component-shade: var(--shade1); /* component header bg */
--border-color: var(--shade2); /* border color for components */
--text-color: var(--shade4); /* main body text */
--text-color-invert: var(--shade0); /* text on dark backgrounds */
--secondary-text-color: var(--shade3); /* lighter body text */
--icon-color: var(--shade5); /* color for svg/ui icons */
